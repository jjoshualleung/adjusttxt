# adjusttxt allows for simple text manipulation of the contents of a file.
# -s <number> --> Skip the lines based on the supplied parameter, number {0, 1}, every even line and odd line respectively
# -w <spacing> --> Remove whitespaces from lines of the input file {"leading", "trailing", "all"} (beginning, middle, end)
# -x --> Remove empty lines - it takes no parameters - mutually exclusive with the -w option
# -r <target> Revers line of the context in the line {words: revers the order of the words, text: reverse each line as whole}
# -p <prefix> Add prefix at the start of the line

# Execution order: -s -> -w or -x --> -r --> -p
# Empty input file shall prodice an empty output
# FILE ERRORS
# 1. The last line of a non-empty input file must be terminated by a newline, if not should generate an error
# 2. The last parameter is always assumed to be a path to the input file, otherwise should generate an error e.g adjusttxt -p "Prefix" <-- without provide file path
# Option ERRORS
# 1. All parameters of program options are required and shall result in an error
# 2. Any unrecognized option shall result in an error


# Categories for adjusttxt parameters

Program:
    Files:
        Empty.  [property EmptyFile]
        Non-empty.  [property NonEmptyFile]
        Last parameter is input file path. [property InputFilePathAtLast]
        Missing new line at the end.    [if NonEmptyFile] [error]
        Missing input file path.    [if !InputFilePathAtLast] [error]

    Option:
        Missing option. # show missing option parameters instead of error
        Unrecognized option.    [error]
        Execution order.

Options:
    Skip lines:
        Skip every even line.   [property EvenLines]
        skip every odd line.    [property OddLines]
        Out-of-range parameter. [error]
        Non-integer.    [error]

    Remove whitespace:
        Leading.    [property LeadingWhitespace]
        Trailing.   [property LeadingWhitespace]
        All.    [property AllWhitespace]
        Unrecognized string.    [error]
        Option with -x. [error]

    Remove empty lines:
        All.    [property RemoveEmptyLines]
        Option with -w. [error]

    Reverse line:
        Words.
        Whole line.
        Unrecognized string. [error] [property InvalidReverseParameter]

    Add prefix:
        Add string.
        Add empty string.   [error]
